**Replace with two-pass approach:**

Pass 1 -- Collect FK data and compute candidate names:
```kotlin
// Collect raw FK data first (without builderFunctionName)
data class RawFk(
    val fkName: String,
    val fkColumnName: String,
    val childFieldExpr: String,
    val parentTableName: String,
    val parentBuilderClassName: String,
    val isSelfRef: Boolean,
    val candidateName: String
)

val rawFks = tableInstance.references
    .filter { fk -> fk.fields.size == 1 }
    .map { fk ->
        val fkColumnName = fk.fields[0].name
        val childFieldExpr = columns.find { it.columnName == fkColumnName }
            ?.tableFieldRefExpression
            ?: "$tableClassName.$tableConstantName.${fkColumnName.uppercase()}"
        val parentTableName = fk.key.table.name
        val parentBuilderClassName = toPascalCase(parentTableName) + "Builder"
        val isSelfRef = parentTableName == tableName
        val strippedFkCol = fkColumnName.removeSuffix("_id")

        val candidateName = if (isSelfRef) {
            toCamelCase(tableName)
        } else if (strippedFkCol == parentTableName) {
            toCamelCase(tableName)       // child table name
        } else {
            toCamelCase(strippedFkCol)   // FK column name fallback
        }

        RawFk(fk.name, fkColumnName, childFieldExpr, parentTableName, parentBuilderClassName, isSelfRef, candidateName)
    }
```

Pass 2 -- Collision detection and final name assignment:
```kotlin
// Find duplicate candidate names
val nameCounts = rawFks.groupingBy { it.candidateName }.eachCount()
val collidingNames = nameCounts.filter { it.value > 1 }.keys

val outboundFKs = rawFks.map { raw ->
    val finalName = if (raw.candidateName in collidingNames) {
        toCamelCase(raw.fkColumnName.removeSuffix("_id"))
    } else {
        raw.candidateName
    }
    ForeignKeyIR(
        fkName = raw.fkName,
        childTableName = tableName,
        childFieldExpression = raw.childFieldExpr,
        parentTableName = raw.parentTableName,
        parentBuilderClassName = raw.parentBuilderClassName,
        builderFunctionName = finalName,
        isSelfReferential = raw.isSelfRef
    )
}
```

The `RawFk` data class can be a private inner class or a local data class inside `extract()`. Keep it local to avoid polluting the public API.

**Key naming rules (from CONTEXT.md decisions):**
- Self-ref: `toCamelCase(tableName)` -- e.g., `parent_id` on `category` -> `"category"`
- FK col stripped matches parent table name: `toCamelCase(childTableName)` -- e.g., `organization_id` on `app_user` where parent is `organization` -> stripped `organization` == `organization` -> use child table name `"appUser"`
- FK col stripped does NOT match parent: `toCamelCase(strippedFkCol)` -- e.g., `created_by` on `task` where parent is `app_user` -> stripped `created_by` != `app_user` -> `"createdBy"`
- Collision: if two FKs produce same candidate, both use `toCamelCase(fkColumnName.removeSuffix("_id"))` -- e.g., two FKs both producing `"user"` -> fall back to FK col names

**Comparison rule:** Always exact snake_case comparison: `fkColumnName.removeSuffix("_id") == parentTableName`. No camelCase transform before comparing.
